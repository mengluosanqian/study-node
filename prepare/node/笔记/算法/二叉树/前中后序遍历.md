### 前序遍历
```js
function preorderTraversal( root ) {
    // write code here
    console.log(root)
    const res = []
    preorder(root,res)
    return res
}
function preorder(root,res) {
    if(!root) return []
    res.push(root.val)
    preorder(root.left,res)
    preorder(root.right,res)
}
```

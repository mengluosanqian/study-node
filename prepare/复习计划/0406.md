# 手写
1. 防抖
```js
        function debounce(fn, delay) {
            let timer = null
            return function () {
                if (timer) {
                    clearTimeout(timer)
                }
                let context = this,
                arg = arguments
                timer = setTimeout(() => {
                    fn.apply(context,arguments)
                    // timer = null
                }, delay)
            }
        }
        let inp = document.querySelector("input")
        inp.addEventListener('input',debounce(function () {
            console.log(this.value);
        }, 500))
```
2. 节流
```js
        function throttle(fn,delay) { 
            let timer = null 
            return function() {
                let context = this,
                arg = arguments
                if(!timer) {
                    timer = setTimeout(() => {
                        fn.apply(context,arguments)
                        timer = null
                    }, delay)
                }
            }
        }
        window.onscroll = throttle(function () {
            console.log('测试节流');
        }, 500)
```

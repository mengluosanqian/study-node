# 手写题
1. 闭包
```js
a = foo();b = foo();c = foo();
// a === 1;b === 2;c === 3;
foo.clear();d = foo(); //d === 1;
function indexAdd() {
  let index = 1
  function foo() {
    return index++
  }
  foo.clear = function() {
    index = 1
  }
  return foo
} 
let foo = indexAdd()
a = foo();b = foo();c = foo();
console.log(a,b,c);
// a === 1;b === 2;c === 3;
foo.clear();d = foo(); //d === 1;
console.log(d);
```
